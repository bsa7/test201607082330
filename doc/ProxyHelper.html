def load_new_proxies
  txt_file_links = parse_proxy_list_links || []
  proxies = txt_file_links.map do |file_link|
    url = "http://webanetlabs.net#{file_link}"
    parse_proxy_list_text(page_load(url: url, check_stamp: /(\d+\.){3}\d+:\d+/))
  end
  proxies.flatten
end
